 13 /**
 14  * <code class="code prettyprint lang-js">createScript</code> compiles <code class="code prettyprint lang-js">code</code> as if it were loaded from <code class="code prettyprint lang-js">filename</code>,
 15  * but does not run it. Instead, it returns a <code class="code prettyprint lang-js">vm.Script</code> object representing this compiled code.
 16  * This script can be run later many times using methods below.
 17  * The returned script is not bound to any global object.
 18  * It is bound before each run, just for that run. <code class="code prettyprint lang-js">filename</code> is optional.
 19  *
 20  * In case of syntax error in <code class="code prettyprint lang-js">code</code>, <code class="code prettyprint lang-js">createScript</code> prints the syntax error to stderr
 21  * and throws an exception.
 22  * @param {*} code
 23  * @param {*} ctx
 24  * @param {*} name
 25  * @return {*}
 26  */
 27 node.vm.prototype.createScript = function(code, ctx, name) {
 28   return node.vm.core_.createScript(code, ctx, name);
 29 };

 45 /**
 46  * Similar to <code class="code prettyprint lang-js">vm.runInThisContext</code> but a method of a precompiled <code class="code prettyprint lang-js">Script</code> object.
 47  * <code class="code prettyprint lang-js">script.runInThisContext</code> runs the code of <code class="code prettyprint lang-js">script</code> and returns the result.
 48  * Running code does not have access to local scope, but does have access to the <code class="code prettyprint lang-js">global</code> object
 49  * (v8: in actual context).
 50  *
 51  * Example of using <code class="code prettyprint lang-js">script.runInThisContext</code> to compile code once and run it multiple times:
 52  * <pre class="code prettyprint lang-js">
 53  *     var vm = require('vm');
 54  *
 55  *     globalVar = 0;
 56  *
 57  *     var script = vm.createScript('globalVar += 1', 'myfile.vm');
 58  *
 59  *     for (var i = 0; i < 1000 ; i += 1) {
 60  *       script.runInThisContext();
 61  *     }
 62  *
 63  *     console.log(globalVar);
 64  *
 65  *     // 1000
 66  * </pre>
 67  * @return {*}
 68  */
 69 node.vm.prototype.runInThisContext = function() {
 70   return node.vm.core_.runInThisContext();
 71 };

 73 /**
 74  * Similar to <code class="code prettyprint lang-js">vm.runInNewContext</code> a method of a precompiled <code class="code prettyprint lang-js">Script</code> object.
 75  * <code class="code prettyprint lang-js">script.runInNewContext</code> runs the code of <code class="code prettyprint lang-js">script</code> with <code class="code prettyprint lang-js">sandbox</code> as the global object and returns the result.
 76  * Running code does not have access to local scope. <code class="code prettyprint lang-js">sandbox</code> is optional.
 77  *
 78  * Example: compile code that increments a global variable and sets one, then execute this code multiple times.
 79  * These globals are contained in the sandbox.
 80  * <pre class="code prettyprint lang-js">
 81  *     var util = require('util'),
 82  *         vm = require('vm'),
 83  *         sandbox = {
 84  *           animal: 'cat',
 85  *           count: 2
 86  *         };
 87  *
 88  *     var script = vm.createScript('count += 1; name = "kitty"', 'myfile.vm');
 89  *
 90  *     for (var i = 0; i < 10 ; i += 1) {
 91  *       script.runInNewContext(sandbox);
 92  *     }
 93  *
 94  *     console.log(util.inspect(sandbox));
 95  *
 96  *     // { animal: 'cat', count: 12, name: 'kitty' }
 97  * </pre>
 98  * Note that running untrusted code is a tricky business requiring great care.  To prevent accidental
 99  * global variable leakage, <code class="code prettyprint lang-js">script.runInNewContext</code> is quite useful, but safely running untrusted code
100  * requires a separate process.
101  * @return {*}
102  */
103 node.vm.prototype.runInNewContext = function() {
104   return node.vm.core_.runInNewContext();
105 };
