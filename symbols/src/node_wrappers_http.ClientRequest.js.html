1 /**
  2  * @name node.http.ClientRequest
  3  * @namespace
  4  * This object is created internally and returned from `http.request()`.  It
  5  * represents an _in-progress_ request whose header has already been queued.  The
  6  * header is still mutable using the `setHeader(name, value)`, `getHeader(name)`,
  7  * `removeHeader(name)` API.  The actual header will be sent along with the first
  8  * data chunk or when closing the connection.
  9  *
 10  * To get the response, add a listener for `'response'` to the request object.
 11  * `'response'` will be emitted from the request object when the response
 12  * headers have been received.  The `'response'` event is executed with one
 13  * argument which is an instance of `http.ClientResponse`.
 14  *
 15  * During the `'response'` event, one can add listeners to the
 16  * response object; particularly to listen for the `'data'` event. Note that
 17  * the `'response'` event is called before any part of the response body is received,
 18  * so there is no need to worry about racing to catch the first part of the
 19  * body. As long as a listener for `'data'` is added during the `'response'`
 20  * event, the entire body will be caught.
 21  *
 22  * <pre class="code prettyprint lang-js">
 23  *     // Good
 24  *     request.on('response', function (response) {
 25  *       response.on('data', function (chunk) {
 26  *         console.log('BODY: ' + chunk);
 27  *       });
 28  *     });
 29  *
 30  *     // Bad - misses all or part of the body
 31  *     request.on('response', function (response) {
 32  *       setTimeout(function () {
 33  *         response.on('data', function (chunk) {
 34  *           console.log('BODY: ' + chunk);
 35  *         });
 36  *       }, 10);
 37  *     });
 38  * </pre>
 39  * This is a `Writable Stream`.
 40  *
 41  * This is an `EventEmitter` with the following events:
 42  */
 43 
 44 goog.provide("node.http.ClientRequest");
