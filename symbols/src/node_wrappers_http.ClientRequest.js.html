1 /**
  2  * @name node.http.ClientRequest
  3  * @namespace
  4  * This object is created internally and returned from <code class="code prettyprint lang-js">http.request()</code>.  It
  5  * represents an <em>in-progress</em> request whose header has already been sent.
  6  *
  7  * To get the response, add a listener for <code class="code prettyprint lang-js">'response'</code> to the request object.
  8  * <code class="code prettyprint lang-js">'response'</code> will be emitted from the request object when the response
  9  * headers have been received.  The <code class="code prettyprint lang-js">'response'</code> event is executed with one
 10  * argument which is an instance of <code class="code prettyprint lang-js">http.ClientResponse</code>.
 11  *
 12  * During the <code class="code prettyprint lang-js">'response'</code> event, one can add listeners to the
 13  * response object; particularly to listen for the <code class="code prettyprint lang-js">'data'</code> event. Note that
 14  * the <code class="code prettyprint lang-js">'response'</code> event is called before any part of the response body is received,
 15  * so there is no need to worry about racing to catch the first part of the
 16  * body. As long as a listener for <code class="code prettyprint lang-js">'data'</code> is added during the <code class="code prettyprint lang-js">'response'</code>
 17  * event, the entire body will be caught.
 18  *
 19  * <pre class="code prettyprint lang-js">
 20  *     // Good
 21  *     request.on('response', function (response) {
 22  *       response.on('data', function (chunk) {
 23  *         console.log('BODY: ' + chunk);
 24  *       });
 25  *     });
 26  *
 27  *     // Bad - misses all or part of the body
 28  *     request.on('response', function (response) {
 29  *       setTimeout(function () {
 30  *         response.on('data', function (chunk) {
 31  *           console.log('BODY: ' + chunk);
 32  *         });
 33  *       }, 10);
 34  *     });
 35  * </pre>
 36  * This is a <code class="code prettyprint lang-js">Writable Stream</code>.
 37  *
 38  * This is an <code class="code prettyprint lang-js">EventEmitter</code> with the following events:
 39  */
 40
