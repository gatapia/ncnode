1 /**

 10 /**
 11  * @param {*} s
 12  * @param {*} decodeSpaces
 13  * @return {*}
 14  */

 17 };
 18

 51 /**
 52  * Serialize an object to a query string.
 53  * Optionally override the default separator and assignment characters.
 54  *
 55  * Example:
 56  * <pre class="code prettyprint lang-js">
 57  *     querystring.stringify({foo: 'bar'})
 58  *     // returns
 59  *     'foo=bar'
 60  *
 61  *     querystring.stringify({foo: 'bar', baz: 'bob'}, ';', ':')
 62  *     // returns
 63  *     'foo:bar;baz:bob'
 64  * </pre>
 65  * @param {*} obj
 66  * @param {*} sep
 67  * @param {*} eq
 68  * @param {*} name
 69  * @return {*}
 70  */
 71 node.querystring.prototype.stringify = function(obj, sep, eq, name) {
 72   return node.querystring.core_.stringify(obj, sep, eq, name);
 73 };
 74

 79  * @return {*}

 85 /**
 86  * Deserialize a query string to an object.
 87  * Optionally override the default separator and assignment characters.
 88  *
 89  * Example:
 90  * <pre class="code prettyprint lang-js">
 91  *     querystring.parse('a=b&b=c')
 92  *     // returns
 93  *     { a: 'b', b: 'c' }
 94  * </pre>
 95  * @param {*} qs
 96  * @param {*} sep
 97  * @param {*} eq
 98  * @return {*}
 99  */
100 node.querystring.prototype.parse = function(qs, sep, eq) {
101   return node.querystring.core_.parse(qs, sep, eq);
102 };
