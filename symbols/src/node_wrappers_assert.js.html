1 /**
  2  * @name node.assert
  3  * @namespace
  4  * This module is used for writing unit tests for your applications, you can
  5  * access it with `require('assert')`.
  6  */
  7 
  8 goog.provide("node.assert");
  9 
 10 /**
 11  * Tests if `actual` is equal to `expected` using the operator provided.
 12  * @param {*} actual
 13  * @param {*} expected
 14  * @param {*} message
 15  * @param {*} operator
 16  * @param {*} stackStartFunction
 17  * @return {*}
 18  */
 19 node.assert.prototype.fail = function(actual, expected, message, operator, stackStartFunction) {
 20   return node.assert.core_.fail(actual, expected, message, operator, stackStartFunction);
 21 };
 22 
 23 /**

 29 node.assert.prototype.ok = function(value, message) {

 99 /**
100  * Expects `block` to throw an error. `error` can be constructor, regexp or
101  * validation function.
102  *
103  * Validate instanceof using constructor:
104  * <pre class="code prettyprint lang-js">
105  *     assert.throws(
106  *       function() {
107  *         throw new Error("Wrong value");
108  *       },
109  *       Error
110  *     );
111  * </pre>
112  * Validate error message using RegExp:
113  * <pre class="code prettyprint lang-js">
114  *     assert.throws(
115  *       function() {
116  *         throw new Error("Wrong value");
117  *       },
118  *       /value/
119  *     );
120  * </pre>
121  * Custom error validation:
122  * <pre class="code prettyprint lang-js">
123  *     assert.throws(
124  *       function() {
125  *         throw new Error("Wrong value");
126  *       },
127  *       function(err) {
128  *         if ( (err instanceof Error) && /value/.test(err) ) {
129  *           return true;
130  *         }
131  *       },
132  *       "unexpected error"
133  *     );
134  * </pre>
135  * @param {*} block
136  * @param {*=} error
137  * @param {*=} message
138  * @return {*}
139  */
140 node.assert.prototype.throws = function(block, error, message) {
141   return node.assert.core_.throws(block, error, message);
142 };
143

151 node.assert.prototype.doesNotThrow = function(block, error, message) {
