1 /**
  2  * @name node.crypto
  3  * @namespace
  4  * Use `require('crypto')` to access this module.
  5  *
  6  * The crypto module requires OpenSSL to be available on the underlying platform.
  7  * It offers a way of encapsulating secure credentials to be used as part
  8  * of a secure HTTPS net or http connection.
  9  *
 10  * It also offers a set of wrappers for OpenSSL's hash, hmac, cipher, decipher, sign and verify methods.
 11  */
 12 
 13 goog.provide("node.crypto");
 14 
 15 /**
 16  * Creates a credentials object, with the optional details being a dictionary with keys:
 17  *
 18  * * `key` : a string holding the PEM encoded private key
 19  * * `cert` : a string holding the PEM encoded certificate
 20  * * `ca` : either a string or list of strings of PEM encoded CA certificates to trust.
 21  *
 22  * If no 'ca' details are given, then node.js will use the default publicly trusted list of CAs as given in
 23  * <http://mxr.mozilla.org/mozilla/source/security/nss/lib/ckfw/builtins/certdata.txt>.
 24  * @param {*} options
 25  * @return {*}
 26  */
 27 node.crypto.prototype.createCredentials = function(options) {
 28   return node.crypto.core_.createCredentials(options);
 29 };
 30 
 31 /**
 32  * Creates and returns a hash object, a cryptographic hash with the given algorithm
 33  * which can be used to generate hash digests.
 34  *
 35  * `algorithm` is dependent on the available algorithms supported by the version
 36  * of OpenSSL on the platform. Examples are `'sha1'`, `'md5'`, `'sha256'`, `'sha512'`, etc.
 37  * On recent releases, `openssl list-message-digest-algorithms` will display the available digest algorithms.
 38  * @param {*} hash
 39  * @return {*}
 40  */
 41 node.crypto.prototype.createHash = function(hash) {
 42   return node.crypto.core_.createHash(hash);
 43 };
 44 
 45 /**
 46  * Creates and returns a hmac object, a cryptographic hmac with the given algorithm and key.
 47  *
 48  * `algorithm` is dependent on the available algorithms supported by OpenSSL - see createHash above.
 49  * `key` is the hmac key to be used.
 50  * @param {*} hmac
 51  * @param {*} key
 52  * @return {*}
 53  */
 54 node.crypto.prototype.createHmac = function(hmac, key) {
 55   return node.crypto.core_.createHmac(hmac, key);
 56 };
 57 
 58 /**
 59  * Creates and returns a cipher object, with the given algorithm and key.
 60  *
 61  * `algorithm` is dependent on OpenSSL, examples are `'aes192'`, etc.
 62  * On recent releases, `openssl list-cipher-algorithms` will display the available cipher algorithms.
 63  * @param {*} cipher
 64  * @param {*} key
 65  * @return {*}
 66  */
 67 node.crypto.prototype.createCipher = function(cipher, key) {
 68   return node.crypto.core_.createCipher(cipher, key);
 69 };
 70 
 71 /**
 72  * @param {*} cipher
 73  * @param {*} key
 74  * @param {*} iv
 75  * @return {*}
 76  */
 77 node.crypto.prototype.createCipheriv = function(cipher, key, iv) {
 78   return node.crypto.core_.createCipheriv(cipher, key, iv);
 79 };
 80 
 81 /**
 82  * Creates and returns a decipher object, with the given algorithm and key.
 83  * This is the mirror of the cipher object above.
 84  * @param {*} cipher
 85  * @param {*} key
 86  * @return {*}
 87  */
 88 node.crypto.prototype.createDecipher = function(cipher, key) {
 89   return node.crypto.core_.createDecipher(cipher, key);
 90 };
 91 
 92 /**
 93  * @param {*} cipher
 94  * @param {*} key
 95  * @param {*} iv
 96  * @return {*}
 97  */
 98 node.crypto.prototype.createDecipheriv = function(cipher, key, iv) {
 99   return node.crypto.core_.createDecipheriv(cipher, key, iv);
100 };
101 
102 /**
103  * Creates and returns a signing object, with the given algorithm.
104  * On recent OpenSSL releases, `openssl list-public-key-algorithms` will display
105  * the available signing algorithms. Examples are `'RSA-SHA256'`.
106  * @param {*} algorithm
107  * @return {*}
108  */
109 node.crypto.prototype.createSign = function(algorithm) {
110   return node.crypto.core_.createSign(algorithm);
111 };
112 
113 /**
114  * Creates and returns a verification object, with the given algorithm.
115  * This is the mirror of the signing object above.
116  * @param {*} algorithm
117  * @return {*}
118  */
119 node.crypto.prototype.createVerify = function(algorithm) {
120   return node.crypto.core_.createVerify(algorithm);
121 };
122 
123 
124 /**
125  * @private
126  * @type {*}
127  */
128 node.crypto.core_ = require("crypto");
