 10 /**
 11  * Resolves <code class="code prettyprint lang-js">to</code> to an absolute path.
 12  *
 13  * If <code class="code prettyprint lang-js">to</code> isn't already absolute <code class="code prettyprint lang-js">from</code> arguments are prepended in right to left
 14  * order, until an absolute path is found. If after using all <code class="code prettyprint lang-js">from</code> paths still
 15  * no absolute path is found, the current working directory is used as well. The
 16  * resulting path is normalized, and trailing slashes are removed unless the path
 17  * gets resolved to the root directory.
 18  *
 19  * Another way to think of it is as a sequence of <code class="code prettyprint lang-js">cd</code> commands in a shell.
 20  * <pre class="code prettyprint lang-js">
 21  *     path.resolve('foo/bar', '/tmp/file/', '..', 'a/../subfile')
 22  * </pre>
 23  * Is similar to:
 24  * <pre class="code prettyprint lang-js">
 25  *     cd foo/bar
 26  *     cd /tmp/file/
 27  *     cd ..
 28  *     cd a/../subfile
 29  *     pwd
 30  * </pre>
 31  * The difference is that the different paths don't need to exist and may also be
 32  * files.
 33  *
 34  * Examples:
 35  * <pre class="code prettyprint lang-js">
 36  *     path.resolve('/foo/bar', './baz')
 37  *     // returns
 38  *     '/foo/bar/baz'
 39  *
 40  *     path.resolve('/foo/bar', '/tmp/file/')
 41  *     // returns
 42  *     '/tmp/file'
 43  *
 44  *     path.resolve('wwwroot', 'static<em>files/png/', '../gif/image.gif')
 45  *     // if currently in /home/myself/node, it returns
 46  *     '/home/myself/node/wwwroot/static</em>files/gif/image.gif'
 47  * </pre>
 48  * @return {*}
 49  */
 50 node.path.prototype.resolve = function() {
 51   return node.path.core_.resolve();
 52 };
 53 
 54 /**
 55  * Normalize a string path, taking care of <code class="code prettyprint lang-js">'..'</code> and <code class="code prettyprint lang-js">'.'</code> parts.
 56  *
 57  * When multiple slashes are found, they're replaces by a single one;
 58  * when the path contains a trailing slash, it is preserved.
 59  * On windows backslashes are used.
 60  *
 61  * Example:
 62  * <pre class="code prettyprint lang-js">
 63  *     path.normalize('/foo/bar//baz/asdf/quux/..')
 64  *     // returns
 65  *     '/foo/bar/baz/asdf'
 66  * </pre>
 67  * @param {*} path
 68  * @return {*}
 69  */
 70 node.path.prototype.normalize = function(path) {
 71   return node.path.core_.normalize(path);
 72 };
 73 
 74 /**
 75  * Join all arguments together and normalize the resulting path.
 76  *
 77  * Example:
 78  * <pre class="code prettyprint lang-js">
 79  *     node> require('path').join(
 80  *     ...   '/foo', 'bar', 'baz/asdf', 'quux', '..')
 81  *     '/foo/bar/baz/asdf'
 82  * </pre>
 83  * @return {*}
 84  */
 85 node.path.prototype.join = function() {
 86   return node.path.core_.join();
 87 };

 89 /**
 90  * Return the directory name of a path.  Similar to the Unix <code class="code prettyprint lang-js">dirname</code> command.
 91  *
 92  * Example:
 93  * <pre class="code prettyprint lang-js">
 94  *     path.dirname('/foo/bar/baz/asdf/quux')
 95  *     // returns
 96  *     '/foo/bar/baz/asdf'
 97  * </pre>
 98  * @param {*} path
 99  * @return {*}
100  */
101 node.path.prototype.dirname = function(path) {
102   return node.path.core_.dirname(path);
103 };
104 
105 /**
106  * Return the last portion of a path.  Similar to the Unix <code class="code prettyprint lang-js">basename</code> command.
107  *
108  * Example:
109  * <pre class="code prettyprint lang-js">
110  *     path.basename('/foo/bar/baz/asdf/quux.html')
111  *     // returns
112  *     'quux.html'
113  *
114  *     path.basename('/foo/bar/baz/asdf/quux.html', '.html')
115  *     // returns
116  *     'quux'
117  * </pre>
118  * @param {*} path
119  * @param {*} ext
120  * @return {*}
121  */
122 node.path.prototype.basename = function(path, ext) {
123   return node.path.core_.basename(path, ext);
124 };
125 
126 /**
127  * Return the extension of the path.  Everything after the last '.' in the last portion
128  * of the path.  If there is no '.' in the last portion of the path or the only '.' is
129  * the first character, then it returns an empty string.  Examples:
130  * <pre class="code prettyprint lang-js">
131  *     path.extname('index.html')
132  *     // returns
133  *     '.html'
134  *
135  *     path.extname('index')
136  *     // returns
137  *     ''
138  * </pre>
139  * @param {*} path
140  * @return {*}
141  */
142 node.path.prototype.extname = function(path) {
143   return node.path.core_.extname(path);
144 };
145 
146 /**
147  * Test whether or not the given path exists.  Then, call the <code class="code prettyprint lang-js">callback</code> argument
148  * with either true or false. Example:
149  * <pre class="code prettyprint lang-js">
150  *     path.exists('/etc/passwd', function (exists) {
151  *       util.debug(exists ? "it's there" : "no passwd!");
152  *     });
153  * @param {*} path
154  * @param {*} callback
155  * @return {*}
156  */
157 node.path.prototype.exists = function(path, callback) {
158   return node.path.core_.exists(path, callback);
159 };

162  * @param {*} path
163  * @return {*}
164  */
165 node.path.prototype.existsSync = function(path) {
166   return node.path.core_.existsSync(path);
167 };
