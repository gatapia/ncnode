1 /**
  2  * @name node.tls
  3  * @namespace
  4  * Use <code class="code prettyprint lang-js">require('tls')</code> to access this module.
  5  *
  6  * The <code class="code prettyprint lang-js">tls</code> module uses OpenSSL to provide Transport Layer Security and/or
  7  * Secure Socket Layer: encrypted stream communication.
  8  *
  9  * TLS/SSL is a public/private key infrastructure. Each client and each
 10  * server must have a private key. A private key is created like this
 11  * <pre class="code prettyprint lang-js">
 12  *     openssl genrsa -out ryans-key.pem 1024
 13  * </pre>
 14  * All severs and some clients need to have a certificate. Certificates are public
 15  * keys signed by a Certificate Authority or self-signed. The first step to
 16  * getting a certificate is to create a "Certificate Signing Request" (CSR)
 17  * file. This is done with:
 18  * <pre class="code prettyprint lang-js">
 19  *     openssl req -new -key ryans-key.pem -out ryans-csr.pem
 20  * </pre>
 21  * To create a self-signed certificate with the CSR, do this:
 22  * <pre class="code prettyprint lang-js">
 23  *     openssl x509 -req -in ryans-csr.pem -signkey ryans-key.pem -out ryans-cert.pem
 24  * </pre>
 25  * Alternatively you can send the CSR to a Certificate Authority for signing.
 26  *
 27  * (TODO: docs on creating a CA, for now interested users should just look at
 28  * <code class="code prettyprint lang-js">test/fixtures/keys/Makefile</code> in the Node source code)
 29  */
 30 
 31 goog.provide("node.tls");

 53 /**
 54  * Creates a new client connection to the given <code class="code prettyprint lang-js">port</code> and <code class="code prettyprint lang-js">host</code>. (If <code class="code prettyprint lang-js">host</code>
 55  * defaults to <code class="code prettyprint lang-js">localhost</code>.) <code class="code prettyprint lang-js">options</code> should be an object which specifies
 56  *
 57  *   - <code class="code prettyprint lang-js">key</code>: A string or <code class="code prettyprint lang-js">Buffer</code> containing the private key of the server in
 58  *     PEM format. (Required)
 59  *
 60  *   - <code class="code prettyprint lang-js">cert</code>: A string or <code class="code prettyprint lang-js">Buffer</code> containing the certificate key of the server in
 61  *     PEM format.
 62  *
 63  *   - <code class="code prettyprint lang-js">ca</code>: An array of strings or <code class="code prettyprint lang-js">Buffer</code>s of trusted certificates. If this is
 64  *     omitted several well known "root" CAs will be used, like VeriSign.
 65  *     These are used to authorize connections.
 66  *
 67  * <code class="code prettyprint lang-js">tls.connect()</code> returns a cleartext <code class="code prettyprint lang-js">CryptoStream</code> object.
 68  *
 69  * After the TLS/SSL handshake the <code class="code prettyprint lang-js">callback</code> is called. The <code class="code prettyprint lang-js">callback</code> will be
 70  * called no matter if the server's certificate was authorized or not. It is up
 71  * to the user to test <code class="code prettyprint lang-js">s.authorized</code> to see if the server certificate was
 72  * signed by one of the specified CAs. If <code class="code prettyprint lang-js">s.authorized === false</code> then the error
 73  * can be found in <code class="code prettyprint lang-js">s.authorizationError</code>.
 74  * @param {*} port /* host
 75  * @param {*} options
 76  * @param {*} cb */
 77  * @return {*}
 78  */
 79 node.tls.prototype.connect = function(port /* host, options, cb */) {
 80   return node.tls.core_.connect(port /* host, options, cb */);
 81 };
