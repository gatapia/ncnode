 45 /**
 46  * Return a string representation of <code>object</code>, which is useful for debugging.
 47  *
 48  * If <code>showHidden</code> is <code>true</code>, then the object's non-enumerable properties will be
 49  * shown too.
 50  *
 51  * If <code>depth</code> is provided, it tells <code>inspect</code> how many times to recurse while
 52  * formatting the object. This is useful for inspecting large complicated objects.
 53  *
 54  * The default is to only recurse twice.  To make it recurse indefinitely, pass
 55  * in <code>null</code> for <code>depth</code>.
 56  *
 57  * Example of inspecting all properties of the <code>util</code> object:
 58  * <pre>
 59  *     var util = require('util');
 60  *
 61  *     console.log(util.inspect(util, true, null));
 62  * </pre>
 63  * @param {*} obj
 64  * @param {*} showHidden
 65  * @param {*} depth
 66  * @param {*} colors
 67  * @return {*}
 68  */
 69 node.util.prototype.inspect = function(obj, showHidden, depth, colors) {
 70   return node.util.core_.inspect(obj, showHidden, depth, colors);
 71 };

116 /**
117  * Inherit the prototype methods from one
118  * <a href="node.https://developer.mozilla.org/en/JavaScript/Reference/Global<em>Objects/Object/constructor">constructor</a>
119  * into another.  The prototype of <code>constructor</code> will be set to a new
120  * object created from <code>superConstructor</code>.
121  *
122  * As an additional convenience, <code>superConstructor</code> will be accessible
123  * through the <code>constructor.super</em></code> property.
124  * <pre>
125  *     var util = require("util");
126  *     var events = require("events");
127  *
128  *     function MyStream() {
129  *         events.EventEmitter.call(this);
130  *     }
131  *
132  *     util.inherits(MyStream, events.EventEmitter);
133  *
134  *     MyStream.prototype.write = function(data) {
135  *         this.emit("data", data);
136  *     }
137  *
138  *     var stream = new MyStream();
139  *
140  *     console.log(stream instanceof events.EventEmitter); // true
141  *     console.log(MyStream.super_ === events.EventEmitter); // true
142  *
143  *     stream.on("data", function(data) {
144  *         console.log('Received data: "' + data + '"');
145  *     })
146  *     stream.write("It works!"); // Received data: "It works!"
147  * @param {*} ctor
148  * @param {*} superCtor
149  * @return {*}
150  */
151 node.util.prototype.inherits = function(ctor, superCtor) {
152   return node.util.core_.inherits(ctor, superCtor);
153 };
