 50 /**
 51  * Return a string representation of `object`, which is useful for debugging.
 52  *
 53  * If `showHidden` is `true`, then the object's non-enumerable properties will be
 54  * shown too.
 55  *
 56  * If `depth` is provided, it tells `inspect` how many times to recurse while
 57  * formatting the object. This is useful for inspecting large complicated objects.
 58  *
 59  * The default is to only recurse twice.  To make it recurse indefinitely, pass
 60  * in `null` for `depth`.
 61  *
 62  * Example of inspecting all properties of the `util` object:
 63  * <pre>
 64  *     var util = require('util');
 65  *
 66  *     console.log(util.inspect(util, true, null));
 67  * </pre>
 68  * @param {*} obj
 69  * @param {*} showHidden
 70  * @param {*} depth
 71  * @param {*} colors
 72  * @return {*}
 73  */
 74 node.util.prototype.inspect = function(obj, showHidden, depth, colors) {
 75   return node.util.core_.inspect(obj, showHidden, depth, colors);
 76 };

121 /**
122  * Inherit the prototype methods from one
123  * [constructor](https://developer.mozilla.org/en/JavaScript/Reference/Global_Objects/Object/constructor)
124  * into another.  The prototype of `constructor` will be set to a new
125  * object created from `superConstructor`.
126  *
127  * As an additional convenience, `superConstructor` will be accessible
128  * through the `constructor.super_` property.
129  * <pre>
130  *     var util = require("util");
131  *     var events = require("events");
132  *
133  *     function MyStream() {
134  *         events.EventEmitter.call(this);
135  *     }
136  *
137  *     util.inherits(MyStream, events.EventEmitter);
138  *
139  *     MyStream.prototype.write = function(data) {
140  *         this.emit("data", data);
141  *     }
142  *
143  *     var stream = new MyStream();
144  *
145  *     console.log(stream instanceof events.EventEmitter); // true
146  *     console.log(MyStream.super_ === events.EventEmitter); // true
147  *
148  *     stream.on("data", function(data) {
149  *         console.log('Received data: "' + data + '"');
150  *     })
151  *     stream.write("It works!"); // Received data: "It works!"
152  * @param {*} ctor
153  * @param {*} superCtor
154  * @return {*}
155  */
156 node.util.prototype.inherits = function(ctor, superCtor) {
157   return node.util.core_.inherits(ctor, superCtor);
158 };
