1 /**
  2  * @name node.http
  3  * @namespace
  4  * To use the HTTP server and client one must <code>require('http')</code>.
  5  *
  6  * The HTTP interfaces in Node are designed to support many features
  7  * of the protocol which have been traditionally difficult to use.
  8  * In particular, large, possibly chunk-encoded, messages. The interface is
  9  * careful to never buffer entire requests or responses--the
 10  * user is able to stream data.
 11  *
 12  * HTTP message headers are represented by an object like this:
 13  * <pre>
 14  *     { 'content-length': '123',
 15  *       'content-type': 'text/plain',
 16  *       'connection': 'keep-alive',
 17  *       'accept': '*/*' }
 18  * </pre>
 19  * Keys are lowercased. Values are not modified.
 20  *
 21  * In order to support the full spectrum of possible HTTP applications, Node's
 22  * HTTP API is very low-level. It deals with stream handling and message
 23  * parsing only. It parses a message into headers and body but it does not
 24  * parse the actual headers or the body.
 25  */
 26 
 27 goog.provide("node.http");
