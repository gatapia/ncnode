 56 /**
 57  * Serialize an object to a query string.
 58  * Optionally override the default separator and assignment characters.
 59  *
 60  * Example:
 61  * <pre>
 62  *     querystring.stringify({foo: 'bar'})
 63  *     // returns
 64  *     'foo=bar'
 65  *
 66  *     querystring.stringify({foo: 'bar', baz: 'bob'}, ';', ':')
 67  *     // returns
 68  *     'foo:bar;baz:bob'
 69  * </pre>
 70  * @param {*} obj
 71  * @param {*} sep
 72  * @param {*} eq
 73  * @param {*} name
 74  * @return {*}
 75  */
 76 node.querystring.prototype.stringify = function(obj, sep, eq, name) {
 77   return node.querystring.core_.stringify(obj, sep, eq, name);
 78 };
 79 
 80 /**
 81  * @param {*} qs
 82  * @param {*} sep
 83  * @param {*} eq
 84  * @return {*}
 85  */
 86 node.querystring.prototype.decode = function(qs, sep, eq) {
 87   return node.querystring.core_.decode(qs, sep, eq);
 88 };

 90 /**
 91  * Deserialize a query string to an object.
 92  * Optionally override the default separator and assignment characters.
 93  *
 94  * Example:
 95  * <pre>
 96  *     querystring.parse('a=b&b=c')
 97  *     // returns
 98  *     { a: 'b', b: 'c' }
 99  * </pre>
100  * @param {*} qs
101  * @param {*} sep
102  * @param {*} eq
103  * @return {*}
104  */
105 node.querystring.prototype.parse = function(qs, sep, eq) {
106   return node.querystring.core_.parse(qs, sep, eq);
107 };
