 18 /**
 19  * `createScript` compiles `code` as if it were loaded from `filename`,
 20  * but does not run it. Instead, it returns a `vm.Script` object representing this compiled code.
 21  * This script can be run later many times using methods below.
 22  * The returned script is not bound to any global object.
 23  * It is bound before each run, just for that run. `filename` is optional.
 24  *
 25  * In case of syntax error in `code`, `createScript` prints the syntax error to stderr
 26  * and throws an exception.
 27  * @param {*} code
 28  * @param {*} ctx
 29  * @param {*} name
 30  * @return {*}
 31  */
 32 node.vm.prototype.createScript = function(code, ctx, name) {
 33   return node.vm.core_.createScript(code, ctx, name);
 34 };

 50 /**
 51  * Similar to `vm.runInThisContext` but a method of a precompiled `Script` object.
 52  * `script.runInThisContext` runs the code of `script` and returns the result.
 53  * Running code does not have access to local scope, but does have access to the `global` object
 54  * (v8: in actual context).
 55  *
 56  * Example of using `script.runInThisContext` to compile code once and run it multiple times:
 57  * <pre>
 58  *     var vm = require('vm');
 59  *
 60  *     globalVar = 0;
 61  *
 62  *     var script = vm.createScript('globalVar += 1', 'myfile.vm');
 63  *
 64  *     for (var i = 0; i < 1000 ; i += 1) {
 65  *       script.runInThisContext();
 66  *     }
 67  *
 68  *     console.log(globalVar);
 69  *
 70  *     // 1000
 71  * </pre>
 72  * @return {*}
 73  */
 74 node.vm.prototype.runInThisContext = function() {
 75   return node.vm.core_.runInThisContext();
 76 };

 78 /**
 79  * Similar to `vm.runInNewContext` a method of a precompiled `Script` object.
 80  * `script.runInNewContext` runs the code of `script` with `sandbox` as the global object and returns the result.
 81  * Running code does not have access to local scope. `sandbox` is optional.
 82  *
 83  * Example: compile code that increments a global variable and sets one, then execute this code multiple times.
 84  * These globals are contained in the sandbox.
 85  * <pre>
 86  *     var util = require('util'),
 87  *         vm = require('vm'),
 88  *         sandbox = {
 89  *           animal: 'cat',
 90  *           count: 2
 91  *         };
 92  *
 93  *     var script = vm.createScript('count += 1; name = "kitty"', 'myfile.vm');
 94  *
 95  *     for (var i = 0; i < 10 ; i += 1) {
 96  *       script.runInNewContext(sandbox);
 97  *     }
 98  *
 99  *     console.log(util.inspect(sandbox));
100  *
101  *     // { animal: 'cat', count: 12, name: 'kitty' }
102  * </pre>
103  * Note that running untrusted code is a tricky business requiring great care.  To prevent accidental
104  * global variable leakage, `script.runInNewContext` is quite useful, but safely running untrusted code
105  * requires a separate process.
106  * @return {*}
107  */
108 node.vm.prototype.runInNewContext = function() {
109   return node.vm.core_.runInNewContext();
110 };
