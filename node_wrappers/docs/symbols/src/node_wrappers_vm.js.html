 13 /**
 14  * `createScript` compiles `code` as if it were loaded from `filename`,
 15  * but does not run it. Instead, it returns a `vm.Script` object representing this compiled code.
 16  * This script can be run later many times using methods below.
 17  * The returned script is not bound to any global object.
 18  * It is bound before each run, just for that run. `filename` is optional.
 19  *
 20  * In case of syntax error in `code`, `createScript` prints the syntax error to stderr
 21  * and throws an exception.
 22  * @param {*} code
 23  * @param {*} ctx
 24  * @param {*} name
 25  * @return {*}
 26  */
 27 node.vm.prototype.createScript = function(code, ctx, name) {
 28   return node.vm.core_.createScript(code, ctx, name);
 29 };

 45 /**
 46  * Similar to `vm.runInThisContext` but a method of a precompiled `Script` object.
 47  * `script.runInThisContext` runs the code of `script` and returns the result.
 48  * Running code does not have access to local scope, but does have access to the `global` object
 49  * (v8: in actual context).
 50  *
 51  * Example of using `script.runInThisContext` to compile code once and run it multiple times:
 52  * <pre>
 53  *     var vm = require('vm');
 54  *
 55  *     globalVar = 0;
 56  *
 57  *     var script = vm.createScript('globalVar += 1', 'myfile.vm');
 58  *
 59  *     for (var i = 0; i < 1000 ; i += 1) {
 60  *       script.runInThisContext();
 61  *     }
 62  *
 63  *     console.log(globalVar);
 64  *
 65  *     // 1000
 66  * </pre>
 67  * @return {*}
 68  */
 69 node.vm.prototype.runInThisContext = function() {
 70   return node.vm.core_.runInThisContext();
 71 };
 72 
 73 /**
 74  * Similar to `vm.runInNewContext` a method of a precompiled `Script` object.
 75  * `script.runInNewContext` runs the code of `script` with `sandbox` as the global object and returns the result.
 76  * Running code does not have access to local scope. `sandbox` is optional.
 77  *
 78  * Example: compile code that increments a global variable and sets one, then execute this code multiple times.
 79  * These globals are contained in the sandbox.
 80  * <pre>
 81  *     var util = require('util'),
 82  *         vm = require('vm'),
 83  *         sandbox = {
 84  *           animal: 'cat',
 85  *           count: 2
 86  *         };
 87  *
 88  *     var script = vm.createScript('count += 1; name = "kitty"', 'myfile.vm');
 89  *
 90  *     for (var i = 0; i < 10 ; i += 1) {
 91  *       script.runInNewContext(sandbox);
 92  *     }
 93  *
 94  *     console.log(util.inspect(sandbox));
 95  *
 96  *     // { animal: 'cat', count: 12, name: 'kitty' }
 97  * </pre>
 98  * Note that running untrusted code is a tricky business requiring great care.  To prevent accidental
 99  * global variable leakage, `script.runInNewContext` is quite useful, but safely running untrusted code
100  * requires a separate process.
101  * @return {*}
102  */
103 node.vm.prototype.runInNewContext = function() {
104   return node.vm.core_.runInNewContext();
105 };
