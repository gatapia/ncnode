 18  *           if (err) {
 19  *             console.log('reverse for ' + a + ' failed: ' +

 21  *           } else {
 22  *             console.log('reverse for ' + a + ': ' +

 93 /**
 94  * Resolves a domain (e.g. `'google.com'`) into an array of the record types
 95  * specified by rrtype. Valid rrtypes are `A` (IPV4 addresses), `AAAA` (IPV6
 96  * addresses), `MX` (mail exchange records), `TXT` (text records), `SRV` (SRV
 97  * records), and `PTR` (used for reverse IP lookups).
 98  *
 99  * The callback has arguments `(err, addresses)`.  The type of each item
100  * in `addresses` is determined by the record type, and described in the
101  * documentation for the corresponding lookup methods below.
102  *
103  * On error, `err` would be an instanceof `Error` object, where `err.errno` is
104  * one of the error codes listed below and `err.message` is a string describing
105  * the error in English.
106  * @param {*} domain
107  * @param {*} type_
108  * @param {*} callback_
109  * @return {*}
110  */
111 node.dns.prototype.resolve = function(domain, type_, callback_) {
112   return node.dns.core_.resolve(domain, type_, callback_);
113 };

135 /**
136  * Resolves a domain (e.g. `'google.com'`) into the first found A (IPv4) or
137  * AAAA (IPv6) record.
138  *
139  * The callback has arguments `(err, address, family)`.  The `address` argument
140  * is a string representation of a IP v4 or v6 address. The `family` argument
141  * is either the integer 4 or 6 and denotes the family of `address` (not
142  * neccessarily the value initially passed to `lookup`).
143  * @param {*} domain
144  * @param {*} family
145  * @param {*} callback
146  * @return {*}
147  */
148 node.dns.prototype.lookup = function(domain, family, callback) {
149   return node.dns.core_.lookup(domain, family, callback);
150 };

212 /**
213  * Reverse resolves an ip address to an array of domain names.
214  *
215  * The callback has arguments `(err, domains)`.
216  *
217  * If there an an error, `err` will be non-null and an instanceof the Error
218  * object.
219  *
220  * Each DNS query can return an error code.
221  *
222  * - `dns.TEMPFAIL`: timeout, SERVFAIL or similar.
223  * - `dns.PROTOCOL`: got garbled reply.
224  * - `dns.NXDOMAIN`: domain does not exists.
225  * - `dns.NODATA`: domain exists but no data of reqd type.
226  * - `dns.NOMEM`: out of memory while processing.
227  * - `dns.BADQUERY`: the query is malformed.
228  * @param {*} domain
229  * @param {*} callback
230  * @return {*}
231  */
232 node.dns.prototype.reverse = function(domain, callback) {
233   return node.dns.core_.reverse(domain, callback);
234 };
