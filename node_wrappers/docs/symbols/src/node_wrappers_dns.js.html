  4 /**
  5  * @fileoverview Use `require('dns')` to access this module.
  6  *
  7  * Here is an example which resolves `'www.google.com'` then reverse
  8  * resolves the IP addresses which are returned.
  9  * <pre>
 10  *     var dns = require('dns');
 11  *
 12  *     dns.resolve4('www.google.com', function (err, addresses) {
 13  *       if (err) throw err;
 14  *
 15  *       console.log('addresses: ' + JSON.stringify(addresses));
 16  *
 17  *       addresses.forEach(function (a) {
 18  *         dns.reverse(a, function (err, domains) {
 19  *           if (err) {
 20  *             console.log('reverse for ' + a + ' failed: ' +
 21  *               err.message);
 22  *           } else {
 23  *             console.log('reverse for ' + a + ': ' +
 24  *               JSON.stringify(domains));
 25  *           }
 26  *         });
 27  *       });
 28  *     });
 29  * </pre>
 30  */

 98 /**
 99  * Resolves a domain (e.g. `'google.com'`) into an array of the record types
100  * specified by rrtype. Valid rrtypes are `A` (IPV4 addresses), `AAAA` (IPV6
101  * addresses), `MX` (mail exchange records), `TXT` (text records), `SRV` (SRV
102  * records), and `PTR` (used for reverse IP lookups).
103  *
104  * The callback has arguments `(err, addresses)`.  The type of each item
105  * in `addresses` is determined by the record type, and described in the
106  * documentation for the corresponding lookup methods below.
107  *
108  * On error, `err` would be an instanceof `Error` object, where `err.errno` is
109  * one of the error codes listed below and `err.message` is a string describing
110  * the error in English.
111  * @param {*} domain
112  * @param {*} type_
113  * @param {*} callback_
114  * @return {*}
115  */
116 node.dns.prototype.resolve = function(domain, type_, callback_) {
117   return node.dns.core_.resolve(domain, type_, callback_);
118 };

140 /**

143  *
144  * The callback has arguments `(err, address, family)`.  The `address` argument
145  * is a string representation of a IP v4 or v6 address. The `family` argument
146  * is either the integer 4 or 6 and denotes the family of `address` (not
147  * neccessarily the value initially passed to `lookup`).
148  * @param {*} domain
149  * @param {*} family
150  * @param {*} callback
151  * @return {*}
152  */

154   return node.dns.core_.lookup(domain, family, callback);

217 /**
218  * Reverse resolves an ip address to an array of domain names.
219  *
220  * The callback has arguments `(err, domains)`.
221  *
222  * If there an an error, `err` will be non-null and an instanceof the Error
223  * object.
224  *
225  * Each DNS query can return an error code.
226  *
227  * - `dns.TEMPFAIL`: timeout, SERVFAIL or similar.
228  * - `dns.PROTOCOL`: got garbled reply.
229  * - `dns.NXDOMAIN`: domain does not exists.
230  * - `dns.NODATA`: domain exists but no data of reqd type.
231  * - `dns.NOMEM`: out of memory while processing.
232  * - `dns.BADQUERY`: the query is malformed.
233  * @param {*} domain
234  * @param {*} callback
235  * @return {*}
236  */
237 node.dns.prototype.reverse = function(domain, callback) {
238   return node.dns.core_.reverse(domain, callback);
239 };
