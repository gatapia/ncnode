1 /**
  2  * @name node.dns
  3  * @namespace
  4  * Use <code>require('dns')</code> to access this module.
  5  *
  6  * Here is an example which resolves <code>'www.google.com'</code> then reverse
  7  * resolves the IP addresses which are returned.
  8  * <pre>
  9  *     var dns = require('dns');
 10  *
 11  *     dns.resolve4('www.google.com', function (err, addresses) {
 12  *       if (err) throw err;
 13  *
 14  *       console.log('addresses: ' + JSON.stringify(addresses));
 15  *
 16  *       addresses.forEach(function (a) {
 17  *         dns.reverse(a, function (err, domains) {
 18  *           if (err) {
 19  *             console.log('reverse for ' + a + ' failed: ' +
 20  *               err.message);
 21  *           } else {
 22  *             console.log('reverse for ' + a + ': ' +
 23  *               JSON.stringify(domains));
 24  *           }
 25  *         });
 26  *       });
 27  *     });
 28  * </pre>
 29  */

 93 /**
 94  * Resolves a domain (e.g. <code>'google.com'</code>) into an array of the record types
 95  * specified by rrtype. Valid rrtypes are <code>A</code> (IPV4 addresses), <code>AAAA</code> (IPV6
 96  * addresses), <code>MX</code> (mail exchange records), <code>TXT</code> (text records), <code>SRV</code> (SRV
 97  * records), and <code>PTR</code> (used for reverse IP lookups).
 98  *
 99  * The callback has arguments <code>(err, addresses)</code>.  The type of each item
100  * in <code>addresses</code> is determined by the record type, and described in the
101  * documentation for the corresponding lookup methods below.
102  *
103  * On error, <code>err</code> would be an instanceof <code>Error</code> object, where <code>err.errno</code> is
104  * one of the error codes listed below and <code>err.message</code> is a string describing
105  * the error in English.
106  * @param {*} domain
107  * @param {*} type_
108  * @param {*} callback_
109  * @return {*}
110  */
111 node.dns.prototype.resolve = function(domain, type_, callback_) {
112   return node.dns.core_.resolve(domain, type_, callback_);
113 };

135 /**
136  * Resolves a domain (e.g. <code>'google.com'</code>) into the first found A (IPv4) or
137  * AAAA (IPv6) record.
138  *
139  * The callback has arguments <code>(err, address, family)</code>.  The <code>address</code> argument
140  * is a string representation of a IP v4 or v6 address. The <code>family</code> argument
141  * is either the integer 4 or 6 and denotes the family of <code>address</code> (not
142  * neccessarily the value initially passed to <code>lookup</code>).
143  * @param {*} domain
144  * @param {*} family
145  * @param {*} callback
146  * @return {*}
147  */
148 node.dns.prototype.lookup = function(domain, family, callback) {
149   return node.dns.core_.lookup(domain, family, callback);
150 };

212 /**
213  * Reverse resolves an ip address to an array of domain names.
214  *
215  * The callback has arguments <code>(err, domains)</code>.
216  *
217  * If there an an error, <code>err</code> will be non-null and an instanceof the Error
218  * object.
219  *
220  * Each DNS query can return an error code.
221  *
222  * - <code>dns.TEMPFAIL</code>: timeout, SERVFAIL or similar.
223  * - <code>dns.PROTOCOL</code>: got garbled reply.
224  * - <code>dns.NXDOMAIN</code>: domain does not exists.
225  * - <code>dns.NODATA</code>: domain exists but no data of reqd type.
226  * - <code>dns.NOMEM</code>: out of memory while processing.
227  * - <code>dns.BADQUERY</code>: the query is malformed.
228  * @param {*} domain
229  * @param {*} callback
230  * @return {*}
231  */
232 node.dns.prototype.reverse = function(domain, callback) {
233   return node.dns.core_.reverse(domain, callback);
234 };
