356 /**
357  * Send a signal to a process. `pid` is the process id and `signal` is the
358  * string describing the signal to send.  Signal names are strings like
359  * 'SIGINT' or 'SIGUSR1'.  If omitted, the signal will be 'SIGTERM'.
360  * See kill(2) for more information.
361  *
362  * Note that just because the name of this function is `process.kill`, it is
363  * really just a signal sender, like the `kill` system call.  The signal sent
364  * may do something other than kill the target process.
365  *
366  * Example of sending a signal to yourself:
367  * <pre>
368  *     process.on('SIGHUP', function () {
369  *       console.log('Got SIGHUP signal.');
370  *     });
371  *
372  *     setTimeout(function () {
373  *       console.log('Exiting.');
374  *       process.exit(0);
375  *     }, 100);
376  *
377  *     process.kill(process.pid, 'SIGHUP');
378  * </pre>
379  * @param {*} pid
380  * @param {*} sig
381  * @return {*}
382  */
383 node.process.prototype.kill = function(pid, sig) {
384   return node.process.core_.kill(pid, sig);
385 };
