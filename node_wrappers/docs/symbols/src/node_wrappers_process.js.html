351 /**
352  * Send a signal to a process. <code>pid</code> is the process id and <code>signal</code> is the
353  * string describing the signal to send.  Signal names are strings like
354  * 'SIGINT' or 'SIGUSR1'.  If omitted, the signal will be 'SIGTERM'.
355  * See kill(2) for more information.
356  *
357  * Note that just because the name of this function is <code>process.kill</code>, it is
358  * really just a signal sender, like the <code>kill</code> system call.  The signal sent
359  * may do something other than kill the target process.
360  *
361  * Example of sending a signal to yourself:
362  * <pre>
363  *     process.on('SIGHUP', function () {
364  *       console.log('Got SIGHUP signal.');
365  *     });
366  *
367  *     setTimeout(function () {
368  *       console.log('Exiting.');
369  *       process.exit(0);
370  *     }, 100);
371  *
372  *     process.kill(process.pid, 'SIGHUP');
373  * </pre>
374  * @param {*} pid
375  * @param {*} sig
376  * @return {*}
377  */
378 node.process.prototype.kill = function(pid, sig) {
379   return node.process.core_.kill(pid, sig);
380 };
