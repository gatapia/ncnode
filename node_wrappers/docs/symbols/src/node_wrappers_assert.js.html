 15 /**

104 /**
105  * Expects `block` to throw an error. `error` can be constructor, regexp or
106  * validation function.
107  *
108  * Validate instanceof using constructor:
109  * <pre>
110  *     assert.throws(
111  *       function() {
112  *         throw new Error("Wrong value");
113  *       },
114  *       Error
115  *     );
116  * </pre>
117  * Validate error message using RegExp:
118  * <pre>
119  *     assert.throws(
120  *       function() {
121  *         throw new Error("Wrong value");
122  *       },
123  *       /value/
124  *     );
125  * </pre>
126  * Custom error validation:
127  * <pre>
128  *     assert.throws(
129  *       function() {
130  *         throw new Error("Wrong value");
131  *       },
132  *       function(err) {
133  *         if ( (err instanceof Error) && /value/.test(err) ) {
134  *           return true;
135  *         }
136  *       },
137  *       "unexpected error"
138  *     );
139  * </pre>
140  * @param {*} block
141  * @param {*=} error
142  * @param {*=} message
143  * @return {*}
144  */
145 node.assert.prototype.throws = function(block, error, message) {
146   return node.assert.core_.throws(block, error, message);
147 };
148
