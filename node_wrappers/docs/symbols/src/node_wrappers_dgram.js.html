 17  * `udp4`, `udp6`, and `unix_dgram`.
 18  *
 19  * Takes an optional callback which is added as a listener for `message` events.
 20  * @param {*} type
 21  * @param {*} listener
 22  * @return {*}
 23  */
 24 node.dgram.prototype.createSocket = function(type, listener) {

 26 };
