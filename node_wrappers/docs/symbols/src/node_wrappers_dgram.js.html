1 /**
  2  * @name node.dgram
  3  * @namespace
  4  * Datagram sockets are available through `require('dgram')`.  Datagrams are most commonly
  5  * handled as IP/UDP messages but they can also be used over Unix domain sockets.
  6  */
  7 
  8 goog.provide("node.dgram");
  9 
 10 /**
 11  * Creates a datagram socket of the specified types.  Valid types are:
 12  * `udp4`, `udp6`, and `unix_dgram`.
 13  *
 14  * Takes an optional callback which is added as a listener for `message` events.
 15  * @param {*} type
 16  * @param {*} listener
 17  * @return {*}
 18  */
 19 node.dgram.prototype.createSocket = function(type, listener) {
 20   return node.dgram.core_.createSocket(type, listener);
 21 };
 22 
 23 
 24 /**
 25  * @private
 26  * @type {*}
 27  */
 28 node.dgram.core_ = require("dgram");
