1 /**
  2  * @name node.url
  3  * @namespace
  4  * This module has utilities for URL resolution and parsing.
  5  * Call `require('url')` to use it.
  6  *
  7  * Parsed URL objects have some or all of the following fields, depending on
  8  * whether or not they exist in the URL string. Any parts that are not in the URL
  9  * string will not be in the parsed object. Examples are shown for the URL
 10  *
 11  * `'http://user:pass@host.com:8080/p/a/t/h?query=string#hash'`
 12  *
 13  * * `href`: The full URL that was originally parsed.
 14  *
 15  *   Example: `'http://user:pass@host.com:8080/p/a/t/h?query=string#hash'`
 16  * * `protocol`: The request protocol.
 17  *
 18  *   Example: `'http:'`
 19  * * `host`: The full host portion of the URL, including port and authentication information.
 20  *
 21  *   Example: `'user:pass@host.com:8080'`
 22  * * `auth`: The authentication information portion of a URL.
 23  *
 24  *   Example: `'user:pass'`
 25  * * `hostname`: Just the hostname portion of the host.
 26  *
 27  *   Example: `'host.com'`
 28  * * `port`: The port number portion of the host.
 29  *
 30  *   Example: `'8080'`
 31  * * `pathname`: The path section of the URL, that comes after the host and before the query, including the initial slash if present.
 32  *
 33  *   Example: `'/p/a/t/h'`
 34  * * `search`: The 'query string' portion of the URL, including the leading question mark.
 35  *
 36  *   Example: `'?query=string'`
 37  * * `query`: Either the 'params' portion of the query string, or a querystring-parsed object.
 38  *
 39  *   Example: `'query=string'` or `{'query':'string'}`
 40  * * `hash`: The 'fragment' portion of the URL including the pound-sign.
 41  *
 42  *   Example: `'#hash'`
 43  *
 44  * The following methods are provided by the URL module:
 45  */
 46 
 47 goog.provide("node.url");
 48 
 49 /**
 50  * Take a URL string, and return an object.  Pass `true` as the second argument to also parse
 51  * the query string using the `querystring` module.
 52  * @param {*} url
 53  * @param {*} parseQueryString
 54  * @param {*} slashesDenoteHost
 55  * @return {*}
 56  */
 57 node.url.prototype.parse = function(url, parseQueryString, slashesDenoteHost) {
 58   return node.url.core_.parse(url, parseQueryString, slashesDenoteHost);
 59 };
 60 
 61 /**
 62  * Take a base URL, and a href URL, and resolve them as a browser would for an anchor tag.
 63  * @param {*} source
 64  * @param {*} relative
 65  * @return {*}
 66  */
 67 node.url.prototype.resolve = function(source, relative) {
 68   return node.url.core_.resolve(source, relative);
 69 };
 70 
 71 /**
 72  * @param {*} source
 73  * @param {*} relative
 74  * @return {*}
 75  */
 76 node.url.prototype.resolveObject = function(source, relative) {
 77   return node.url.core_.resolveObject(source, relative);
 78 };
 79 
 80 /**
 81  * Take a parsed URL object, and return a formatted URL string.
 82  * @param {*} obj
 83  * @return {*}
 84  */
 85 node.url.prototype.format = function(obj) {
 86   return node.url.core_.format(obj);
 87 };
 88 
 89 
 90 /**
 91  * @private
 92  * @type {*}
 93  */
 94 node.url.core_ = require("url");
