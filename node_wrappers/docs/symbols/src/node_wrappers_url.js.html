  4 /**
  5  * @fileoverview This module has utilities for URL resolution and parsing.
  6  * Call `require('url')` to use it.
  7  *
  8  * Parsed URL objects have some or all of the following fields, depending on
  9  * whether or not they exist in the URL string. Any parts that are not in the URL
 10  * string will not be in the parsed object. Examples are shown for the URL
 11  *
 12  * `'http://user:pass@host.com:8080/p/a/t/h?query=string#hash'`
 13  *
 14  * * `href`: The full URL that was originally parsed.
 15  *
 16  *   Example: `'http://user:pass@host.com:8080/p/a/t/h?query=string#hash'`
 17  * * `protocol`: The request protocol.
 18  *
 19  *   Example: `'http:'`
 20  * * `host`: The full host portion of the URL, including port and authentication information.
 21  *
 22  *   Example: `'user:pass@host.com:8080'`
 23  * * `auth`: The authentication information portion of a URL.
 24  *
 25  *   Example: `'user:pass'`
 26  * * `hostname`: Just the hostname portion of the host.
 27  *
 28  *   Example: `'host.com'`
 29  * * `port`: The port number portion of the host.
 30  *
 31  *   Example: `'8080'`
 32  * * `pathname`: The path section of the URL, that comes after the host and before the query, including the initial slash if present.
 33  *
 34  *   Example: `'/p/a/t/h'`
 35  * * `search`: The 'query string' portion of the URL, including the leading question mark.
 36  *
 37  *   Example: `'?query=string'`
 38  * * `query`: Either the 'params' portion of the query string, or a querystring-parsed object.
 39  *
 40  *   Example: `'query=string'` or `{'query':'string'}`
 41  * * `hash`: The 'fragment' portion of the URL including the pound-sign.
 42  *
 43  *   Example: `'#hash'`
 44  *
 45  * The following methods are provided by the URL module:
 46  */
 47 
 48 /**
 49  * @constructor
 50  * url
 51  */
 52 node.url = function() {};
 53 
 54 /**
 55  * Take a URL string, and return an object.  Pass `true` as the second argument to also parse
 56  * the query string using the `querystring` module.
 57  * @param {*} url
 58  * @param {*} parseQueryString
 59  * @param {*} slashesDenoteHost
 60  * @return {*}
 61  */
 62 node.url.prototype.parse = function(url, parseQueryString, slashesDenoteHost) {
 63   return node.url.core_.parse(url, parseQueryString, slashesDenoteHost);
 64 };
 65 
 66 /**
 67  * Take a base URL, and a href URL, and resolve them as a browser would for an anchor tag.
 68  * @param {*} source
 69  * @param {*} relative
 70  * @return {*}
 71  */
 72 node.url.prototype.resolve = function(source, relative) {
 73   return node.url.core_.resolve(source, relative);
 74 };
 75 
 76 /**
 77  * @param {*} source
 78  * @param {*} relative
 79  * @return {*}
 80  */
 81 node.url.prototype.resolveObject = function(source, relative) {
 82   return node.url.core_.resolveObject(source, relative);
 83 };
 84 
 85 /**
 86  * Take a parsed URL object, and return a formatted URL string.
 87  * @param {*} obj
 88  * @return {*}
 89  */
 90 node.url.prototype.format = function(obj) {
 91   return node.url.core_.format(obj);
 92 };

 95 /**
 96  * @private
 97  * @type {*}
 98  */
 99 node.url.core_ = require("url");
