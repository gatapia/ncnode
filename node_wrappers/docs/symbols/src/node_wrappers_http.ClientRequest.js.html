1 /**
  2  * @name node.http.ClientRequest
  3  * @namespace
  4  * This object is created internally and returned from <code class="code prettyprint lang-js">http.request()</code>.  It
  5  * represents an <em>in-progress</em> request whose header has already been queued.  The
  6  * header is still mutable using the <code class="code prettyprint lang-js">setHeader(name, value)</code>, <code class="code prettyprint lang-js">getHeader(name)</code>,
  7  * <code class="code prettyprint lang-js">removeHeader(name)</code> API.  The actual header will be sent along with the first
  8  * data chunk or when closing the connection.
  9  *
 10  * To get the response, add a listener for <code class="code prettyprint lang-js">'response'</code> to the request object.
 11  * <code class="code prettyprint lang-js">'response'</code> will be emitted from the request object when the response
 12  * headers have been received.  The <code class="code prettyprint lang-js">'response'</code> event is executed with one
 13  * argument which is an instance of <code class="code prettyprint lang-js">http.ClientResponse</code>.
 14  *
 15  * During the <code class="code prettyprint lang-js">'response'</code> event, one can add listeners to the
 16  * response object; particularly to listen for the <code class="code prettyprint lang-js">'data'</code> event. Note that
 17  * the <code class="code prettyprint lang-js">'response'</code> event is called before any part of the response body is received,
 18  * so there is no need to worry about racing to catch the first part of the
 19  * body. As long as a listener for <code class="code prettyprint lang-js">'data'</code> is added during the <code class="code prettyprint lang-js">'response'</code>
 20  * event, the entire body will be caught.
 21  *
 22  * <pre class="code prettyprint lang-js">
 23  *     // Good
 24  *     request.on('response', function (response) {
 25  *       response.on('data', function (chunk) {
 26  *         console.log('BODY: ' + chunk);
 27  *       });
 28  *     });
 29  *
 30  *     // Bad - misses all or part of the body
 31  *     request.on('response', function (response) {
 32  *       setTimeout(function () {
 33  *         response.on('data', function (chunk) {
 34  *           console.log('BODY: ' + chunk);
 35  *         });
 36  *       }, 10);
 37  *     });
 38  * </pre>
 39  * This is a <code class="code prettyprint lang-js">Writable Stream</code>.
 40  *
 41  * This is an <code class="code prettyprint lang-js">EventEmitter</code> with the following events:
 42  */
