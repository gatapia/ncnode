1 /**
  2  * @name node.https
  3  * @namespace
  4  * HTTPS is the HTTP protocol over TLS/SSL. In Node this is implemented as a
  5  * separate module.

 10 /**
 11  * @param {*} opts

 17 };
 18 
 19 /**
 20  * @param {*} options
 21  * @return {*}
 22  */
 23 node.https.prototype.getAgent = function(options) {

 25 };
 26 
 27 /**
 28  * @param {*} options
 29  * @param {*} cb
 30  * @return {*}
 31  */
 32 node.https.prototype.request = function(options, cb) {

 34 };
 35 
 36 /**
 37  * @param {*} options
 38  * @param {*} cb
 39  * @return {*}
 40  */
 41 node.https.prototype.get = function(options, cb) {

 43 };
