1 /**
  2  * @name node.repl
  3  * @namespace
  4  * A Read-Eval-Print-Loop (REPL) is available both as a standalone program and easily
  5  * includable in other programs.  REPL provides a way to interactively run
  6  * JavaScript and see the results.  It can be used for debugging, testing, or
  7  * just trying things out.
  8  *
  9  * By executing `node` without any arguments from the command-line you will be
 10  * dropped into the REPL. It has simplistic emacs line-editing.
 11  * <pre>
 12  *     mjr:~$ node
 13  *     Type '.help' for options.
 14  *     > a = [ 1, 2, 3];
 15  *     [ 1, 2, 3 ]
 16  *     > a.forEach(function (v) {
 17  *     ...   console.log(v);
 18  *     ...   });
 19  *     1
 20  *     2
 21  *     3
 22  * </pre>
 23  * For advanced line-editors, start node with the environmental variable `NODE_NO_READLINE=1`.
 24  * This will start the REPL in canonical terminal settings which will allow you to use with `rlwrap`.
 25  *
 26  * For example, you could add this to your bashrc file:
 27  * <pre>
 28  *     alias node="env NODE_NO_READLINE=1 rlwrap node"
 29  * </pre>
 30  */
 31 
 32 goog.provide("node.repl");
 33 
 34 /**
 35  * @param {*} obj
 36  * @param {*} showHidden
 37  * @param {*} depth
 38  * @param {*} colors
 39  * @return {*}
 40  */
 41 node.repl.prototype.writer = function(obj, showHidden, depth, colors) {
 42   return node.repl.core_.writer(obj, showHidden, depth, colors);
 43 };
 44 
 45 /**
 46  * Starts a REPL with `prompt` as the prompt and `stream` for all I/O.  `prompt`
 47  * is optional and defaults to `> `.  `stream` is optional and defaults to
 48  * `process.stdin`.
 49  *
 50  * Multiple REPLs may be started against the same running instance of node.  Each
 51  * will share the same global object but will have unique I/O.
 52  *
 53  * Here is an example that starts a REPL on stdin, a Unix socket, and a TCP socket:
 54  * <pre>
 55  *     var net = require("net"),
 56  *         repl = require("repl");
 57  *
 58  *     connections = 0;
 59  *
 60  *     repl.start("node via stdin> ");
 61  *
 62  *     net.createServer(function (socket) {
 63  *       connections += 1;
 64  *       repl.start("node via Unix socket> ", socket);
 65  *     }).listen("/tmp/node-repl-sock");
 66  *
 67  *     net.createServer(function (socket) {
 68  *       connections += 1;
 69  *       repl.start("node via TCP socket> ", socket);
 70  *     }).listen(5001);
 71  * </pre>
 72  * Running this program from the command line will start a REPL on stdin.  Other
 73  * REPL clients may connect through the Unix socket or TCP socket. `telnet` is useful
 74  * for connecting to TCP sockets, and `socat` can be used to connect to both Unix and
 75  * TCP sockets.
 76  *
 77  * By starting a REPL from a Unix socket-based server instead of stdin, you can
 78  * connect to a long-running node process without restarting it.
 79  * @param {*} prompt
 80  * @param {*} source
 81  * @return {*}
 82  */
 83 node.repl.prototype.start = function(prompt, source) {
 84   return node.repl.core_.start(prompt, source);
 85 };
 86 
 87 
 88 /**
 89  * @private
 90  * @type {*}
 91  */
 92 node.repl.core_ = require("repl");
