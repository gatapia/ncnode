  4 /**
  5  * @fileoverview Use `require('crypto')` to access this module.
  6  *
  7  * The crypto module requires OpenSSL to be available on the underlying platform.
  8  * It offers a way of encapsulating secure credentials to be used as part
  9  * of a secure HTTPS net or http connection.
 10  *
 11  * It also offers a set of wrappers for OpenSSL's hash, hmac, cipher, decipher, sign and verify methods.
 12  */
 13 
 14 /**
 15  * @constructor
 16  * crypto
 17  */
 18 node.crypto = function() {};
 19 
 20 /**
 21  * Creates a credentials object, with the optional details being a dictionary with keys:
 22  *
 23  * * `key` : a string holding the PEM encoded private key
 24  * * `cert` : a string holding the PEM encoded certificate
 25  * * `ca` : either a string or list of strings of PEM encoded CA certificates to trust.
 26  *
 27  * If no 'ca' details are given, then node.js will use the default publicly trusted list of CAs as given in
 28  * <http://mxr.mozilla.org/mozilla/source/security/nss/lib/ckfw/builtins/certdata.txt>.
 29  * @param {*} options
 30  * @return {*}
 31  */
 32 node.crypto.prototype.createCredentials = function(options) {
 33   return node.crypto.core_.createCredentials(options);
 34 };
 35 
 36 /**
 37  * Creates and returns a hash object, a cryptographic hash with the given algorithm
 38  * which can be used to generate hash digests.
 39  *
 40  * `algorithm` is dependent on the available algorithms supported by the version
 41  * of OpenSSL on the platform. Examples are `'sha1'`, `'md5'`, `'sha256'`, `'sha512'`, etc.
 42  * On recent releases, `openssl list-message-digest-algorithms` will display the available digest algorithms.
 43  * @param {*} hash
 44  * @return {*}
 45  */
 46 node.crypto.prototype.createHash = function(hash) {
 47   return node.crypto.core_.createHash(hash);
 48 };
 49 
 50 /**
 51  * Creates and returns a hmac object, a cryptographic hmac with the given algorithm and key.
 52  *
 53  * `algorithm` is dependent on the available algorithms supported by OpenSSL - see createHash above.
 54  * `key` is the hmac key to be used.
 55  * @param {*} hmac
 56  * @param {*} key
 57  * @return {*}
 58  */
 59 node.crypto.prototype.createHmac = function(hmac, key) {
 60   return node.crypto.core_.createHmac(hmac, key);
 61 };
 62 
 63 /**
 64  * Creates and returns a cipher object, with the given algorithm and key.
 65  *
 66  * `algorithm` is dependent on OpenSSL, examples are `'aes192'`, etc.
 67  * On recent releases, `openssl list-cipher-algorithms` will display the available cipher algorithms.
 68  * @param {*} cipher
 69  * @param {*} key
 70  * @return {*}
 71  */
 72 node.crypto.prototype.createCipher = function(cipher, key) {
 73   return node.crypto.core_.createCipher(cipher, key);
 74 };
 75 
 76 /**
 77  * @param {*} cipher
 78  * @param {*} key
 79  * @param {*} iv
 80  * @return {*}
 81  */
 82 node.crypto.prototype.createCipheriv = function(cipher, key, iv) {
 83   return node.crypto.core_.createCipheriv(cipher, key, iv);
 84 };
 85 
 86 /**
 87  * Creates and returns a decipher object, with the given algorithm and key.
 88  * This is the mirror of the cipher object above.
 89  * @param {*} cipher
 90  * @param {*} key
 91  * @return {*}
 92  */
 93 node.crypto.prototype.createDecipher = function(cipher, key) {
 94   return node.crypto.core_.createDecipher(cipher, key);
 95 };
 96 
 97 /**
 98  * @param {*} cipher
 99  * @param {*} key
100  * @param {*} iv
101  * @return {*}
102  */
103 node.crypto.prototype.createDecipheriv = function(cipher, key, iv) {
104   return node.crypto.core_.createDecipheriv(cipher, key, iv);
105 };
106 
107 /**
108  * Creates and returns a signing object, with the given algorithm.
109  * On recent OpenSSL releases, `openssl list-public-key-algorithms` will display
110  * the available signing algorithms. Examples are `'RSA-SHA256'`.
111  * @param {*} algorithm
112  * @return {*}
113  */
114 node.crypto.prototype.createSign = function(algorithm) {
115   return node.crypto.core_.createSign(algorithm);
116 };
117 
118 /**
119  * Creates and returns a verification object, with the given algorithm.
120  * This is the mirror of the signing object above.
121  * @param {*} algorithm
122  * @return {*}
123  */
124 node.crypto.prototype.createVerify = function(algorithm) {
125   return node.crypto.core_.createVerify(algorithm);
126 };
