  4 /**
  5  * @fileoverview The `net` module provides you with an asynchronous network wrapper. It contains
  6  * methods for creating both servers and clients (called streams). You can include
  7  * this module with `require("net");`
  8  */
  9 
 10 /**
 11  * @constructor
 12  * net
 13  */
 14 node.net = function() {};
 15 
 16 /**
 17  * #### net.isIP(input)
 18  *
 19  * Tests if input is an IP address. Returns 0 for invalid strings,
 20  * returns 4 for IP version 4 addresses, and returns 6 for IP version 6 addresses.
 21  *
 22  *
 23  * #### net.isIPv4(input)
 24  *
 25  * Returns true if input is a version 4 IP address, otherwise returns false.
 26  *
 27  *
 28  * #### net.isIPv6(input)
 29  *
 30  * Returns true if input is a version 6 IP address, otherwise returns false.
 31  * @return {*}
 32  */
 33 node.net.prototype.isIP = function() {
 34   return node.net.core_.isIP();
 35 };
 36 
 37 /**
 38  * @param {*} input
 39  * @return {*}
 40  */
 41 node.net.prototype.isIPv4 = function(input) {
 42   return node.net.core_.isIPv4(input);
 43 };
 44 
 45 /**
 46  * @param {*} input
 47  * @return {*}
 48  */
 49 node.net.prototype.isIPv6 = function(input) {
 50   return node.net.core_.isIPv6(input);
 51 };
 52 
 53 /**
 54  * Construct a new socket object and opens a socket to the given location. When
 55  * the socket is established the `'connect'` event will be emitted.
 56  *
 57  * The arguments for this method change the type of connection:
 58  *
 59  * * `net.createConnection(port, [host])`
 60  *
 61  *   Creates a TCP connection to `port` on `host`. If `host` is omitted, `localhost`
 62  *   will be assumed.
 63  *
 64  * * `net.createConnection(path)`
 65  *
 66  *   Creates unix socket connection to `path`
 67  *
 68  * ---
 69  * @param {*} port
 70  * @param {*} host
 71  * @return {*}
 72  */
 73 node.net.prototype.createConnection = function(port, host) {
 74   return node.net.core_.createConnection(port, host);
 75 };
 76 
 77 /**
 78  * Creates a new TCP server. The `connectionListener` argument is
 79  * automatically set as a listener for the `'connection'` event.
 80  * @return {*}
 81  */
 82 node.net.prototype.createServer = function() {
 83   return node.net.core_.createServer();
 84 };
