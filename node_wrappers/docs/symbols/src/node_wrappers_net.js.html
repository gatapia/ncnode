1 /**
  2  * @name node.net
  3  * @namespace
  4  * The <code>net</code> module provides you with an asynchronous network wrapper. It contains
  5  * methods for creating both servers and clients (called streams). You can include
  6  * this module with <code>require("net");</code>
  7  */
  8 
  9 goog.provide("node.net");
 10 
 11 /**
 12  * #### net.isIP(input)
 13  *
 14  * Tests if input is an IP address. Returns 0 for invalid strings,
 15  * returns 4 for IP version 4 addresses, and returns 6 for IP version 6 addresses.
 16  *
 17  *
 18  * #### net.isIPv4(input)
 19  *
 20  * Returns true if input is a version 4 IP address, otherwise returns false.
 21  *
 22  *
 23  * #### net.isIPv6(input)
 24  *
 25  * Returns true if input is a version 6 IP address, otherwise returns false.
 26  * @return {*}
 27  */
 28 node.net.prototype.isIP = function() {
 29   return node.net.core_.isIP();
 30 };
 31 
 32 /**
 33  * @param {*} input
 34  * @return {*}
 35  */
 36 node.net.prototype.isIPv4 = function(input) {
 37   return node.net.core_.isIPv4(input);
 38 };
 39 
 40 /**
 41  * @param {*} input
 42  * @return {*}
 43  */
 44 node.net.prototype.isIPv6 = function(input) {
 45   return node.net.core_.isIPv6(input);
 46 };
 47 
 48 /**
 49  * Construct a new socket object and opens a socket to the given location. When
 50  * the socket is established the <code>'connect'</code> event will be emitted.
 51  *
 52  * The arguments for this method change the type of connection:
 53  *
 54  * * <code>net.createConnection(port, [host])</code>
 55  *
 56  *   Creates a TCP connection to <code>port</code> on <code>host</code>. If <code>host</code> is omitted, <code>localhost</code>
 57  *   will be assumed.
 58  *
 59  * * <code>net.createConnection(path)</code>
 60  *
 61  *   Creates unix socket connection to <code>path</code>
 62  *
 63  * ---
 64  * @param {*} port
 65  * @param {*} host
 66  * @return {*}
 67  */
 68 node.net.prototype.createConnection = function(port, host) {
 69   return node.net.core_.createConnection(port, host);
 70 };
 71 
 72 /**
 73  * Creates a new TCP server. The <code>connectionListener</code> argument is
 74  * automatically set as a listener for the <code>'connection'</code> event.
 75  * @return {*}
 76  */
 77 node.net.prototype.createServer = function() {
 78   return node.net.core_.createServer();
 79 };

 82 /**
 83  * @private
 84  * @type {*}
 85  */
 86 node.net.core_ = require("net");
